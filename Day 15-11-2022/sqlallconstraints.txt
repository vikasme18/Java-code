use vikasdb;

/* Booking Table */
CREATE TABLE BOOKING(BOOKID INT PRIMARY KEY, SOURCE VARCHAR(30) DEFAULT('Mumbai'), DESTINATION VARCHAR(30) DEFAULT('Pune'), COST INT NOT NULL);
DESC BOOKING;

/* Passenger Table */
CREATE TABLE PASSENGER(PID INT PRIMARY KEY, BOOKID INT UNIQUE KEY NOT NULL, PNAME VARCHAR(30) NOT NULL, MOBILE BIGINT UNIQUE KEY, CHECK(length(MOBILE) = 10), AGE INT NOT NULL, CHECK(AGE>5), FOREIGN KEY(BOOKID) REFERENCES BOOKING(BOOKID));
DESC PASSENGER;
